      addUser(username) {
    var users = this.get('users');
    users.push(username);
    this.set({
      'users': users
    });
    return this.save();
  }

    Remove user from users array and sync with server

      removeUser(username) {
    var users = this.get('users');
    if( _.contains(users, username) ) {
      this.set({
        'users':  _.without(users, username)
      });
      return this.save();
    }
    return false;
  }
}
